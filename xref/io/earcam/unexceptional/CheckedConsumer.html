1   package io.earcam.unexceptional;

3   import java.util.Objects;
4   
5   import javax.annotation.Nonnull;
6   
7   /**
8    * A checked exception version of {@link java.util.function.Consumer}
9    * 
10   * @param <T> the input type of the {@link #accept(Object)} operation
11   * 
12   * @see java.util.function.Consumer
13   */
14  @FunctionalInterface
15  public interface CheckedConsumer<T> {
16  
17  	/**
18       * See {@link java.util.function.Consumer#accept(Object)}
19       * 
20       * @param t the input argument
21  	 * @throws Exception a possible checked exception
22  	 */
23  	public abstract void accept(T t) throws Throwable;  //NOSONAR
24  
25      /**
26       * See {@link java.util.function.Consumer#andThen(Consumer)}
27       * 
28       * @param after the operation to perform after this operation
29       * @return a composed {@code CheckedConsumer} that performs in sequence this
30       * operation followed by the {@code after} operation
31       * 
32       * @throws NullPointerException if {@code after} is {@code null}
33       */
34      public default CheckedConsumer<T> andThen(@Nonnull CheckedConsumer<? super T> after) {
35          Objects.requireNonNull(after);
36          return (T t) -> { 
37  			accept(t); 
38  			after.accept(t); 
39  		};
40      }
41  }
