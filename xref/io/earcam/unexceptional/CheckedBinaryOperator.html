11  /**
12   * A checked parallel of {@link java.util.function.BinaryOperator}
13   * 
14   * @param <T> arguments and return type
15   * 
16   * @see java.util.function.BinaryOperator
17   */
18  @FunctionalInterface
19  public interface CheckedBinaryOperator<T> extends CheckedBiFunction<T, T, T> {
20  
21  	/**
22  	 * See {@link java.util.function.BinaryOperator#minBy(Comparator)}
23  	 *
24  	 * @param <T> arguments and return type
25  	 * 
26  	 * @param comparator the comparator used to determine the minimum.
27  	 * @return a {@link CheckedBinaryOperator}
28  	 * @throws NullPointerException if {@code comparator} is {@code null}
29  	 */
30  	public static <T> CheckedBinaryOperator<T> minBy(@Nonnull CheckedComparator<? super T> comparator)
31  	{
32  		requireNonNull(comparator);
33  		return (CheckedBinaryOperator<T> & Serializable) (a, b) -> comparator.compare(a, b) < 0 ? a : b;
34  	}
35  
36  
37  	/**
38  	 * See {@link java.util.function.BinaryOperator#maxBy(Comparator)}
39  	 * 
40  	 * @param <T> arguments and return type
41  	 * 
42  	 * @param comparator the comparator used to determine the maximum.
43  	 * @return a {@link CheckedBinaryOperator}
44  	 * @throws NullPointerException if {@code comparator} is {@code null}
45  	 */
46  	public static <T> CheckedBinaryOperator<T> maxBy(@Nonnull CheckedComparator<? super T> comparator)
47  	{
48  		Objects.requireNonNull(comparator);
49  		return (CheckedBinaryOperator<T> & Serializable) (a, b) -> comparator.compare(a, b) > 0 ? a : b;
50  	}
51  }
