27  /**
28   * Example usage and comparable with/out-cases for {@link EmeticStream} 
29   */
30  public class EmeticStreamExample {
31  
32  
33  	
34  	/**
35  	 * Demonstrates {@link Files#list} and subsequent stream operations, with;
36  	 * 
37  	 * <ul>
38  	 * <li>Vanilla {@link Stream} API (admittedly overly exaggerated boiler-plating)</li>
39  	 * <li>{@link Exceptional} with {@link Stream} API</li>
40  	 * <li>{@link EmeticStream}</li>
41  	 * </ul>
42  	 */
43  	@Test
44  	public void fileListEquivalence()
45  	{
46  		Path path = get(".", "target");
47  
48  		
49  		// Using vanilla Stream API (in a manner that would distress Dijkstra)
50  		List<String> a;
51  		try {
52  			a = Files.list(path)
53  					.map(Path::toFile)
54  					.map(t -> {
55  						try {
56  							return t.getCanonicalFile();
57  						} catch (IOException e) {
58  							throw new UncheckedIOException(e);
59  						}
60  					})
61  					.map(File::toURI)
62  					.map(t -> {
63  						try {
64  							return t.toURL();
65  						} catch (MalformedURLException e) {
66  							throw new UncheckedIOException(e);
67  						}
68  					})
69  					.map(Object::toString)
70  					.collect(toList());
71  		} catch (IOException e) {
72  			throw new UncheckedIOException(e);
73  		}
74  
75  
76  		// Using static imports from Exceptional
77  		List<String> b = apply(Files::list, path)
78  			.map(Path::toFile)
79  			.map(uncheckFunction(File::getCanonicalFile))
80  			.map(File::toURI)
81  			.map(uncheckFunction(URI::toURL))
82  			.map(Object::toString)
83  			.collect(toList());
84  
85  
86  		// Using EmeticStream (less efficient on GC, more efficient on the eye)
87  		List<String> c = emesis(Files::list, path)
88  			.map(Path::toFile)
89  			.map(File::getCanonicalFile)
90  			.map(File::toURI)
91  			.map(Object::toString)
92  			.collect(toList());
93  
94  
95  		assertThat(a, is( equalTo( b )));
96  		assertThat(b, is( equalTo( c )));
97  	}
98  }
